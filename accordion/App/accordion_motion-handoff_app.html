Implementación Dart (Flutter)

const motionSpringSm = SpringDescription(mass: 1.0, stiffness: 400, damping: 35);

void toggleAccordion({required AnimationController controller, required bool expanded}) {
  if (disableAnimations) { controller.value = expanded ? 1.0 : 0.0; return; }
  controller.animateWith(
    SpringSimulation(motionSpringSm, expanded ? 0.0 : 1.0, expanded ? 1.0 : 0.0, 0.0),
  );
}
