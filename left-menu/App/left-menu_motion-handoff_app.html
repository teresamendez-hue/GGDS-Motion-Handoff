// Drawer — motion-spring-md (m1·s300·d28)
Animate(
  target: isOpen ? 1 : 0,
  effects: [
  SlideEffect(
    begin: const Offset(-1, 0),
    end: Offset.zero,
    curve: GgdsMotion.springMd,
  ),
  FadeEffect(
    begin: 0,
    end: 1,
    curve: GgdsMotion.springMd,
    target: scrimOpacity,
  ),
]);

// Salida drawer — Curves.easeIn ~180ms
SlideEffect(
  begin: Offset.zero,
  end: const Offset(-1, 0),
  duration: const Duration(milliseconds: 180),
  curve: Curves.easeIn,
);

// Navegación por columna — motion-spring-md
Transform.translate(
  offset: Offset(-columnWidth * depth, 0),
  child: columnsTrack,

// Ítem pressed / selected — motion-spring-sm

  duration: const Duration(milliseconds: 220),
  curve: GgdsMotion.springSm,
  color: selected ? selectedBg : Colors.transparent,
);

Token Mapping
Abrir / cerrar drawer · tap ítem hoja · drill (›) · atrás (←): haptic-action-press → haptic-impact-light → HapticFeedback.lightImpact()
Slide de columna · scroll: none
void onMenuItemTap(VoidCallback action) {
  context.haptics.trigger(GgdsHapticSemantic.actionPress);
  action();
}

// Ejemplos de trigger
onOpenDrawer: () => onMenuItemTap(openDrawer);
onDrill: () => onMenuItemTap(() => navigateToColumn(nextDepth));
onBack: () => onMenuItemTap(() => navigateToColumn(depth - 1));
onLeafSelect: () => onMenuItemTap(() => selectItem(item));
