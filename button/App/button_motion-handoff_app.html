Implementación Dart (Flutter)

final motion = context.motionTokens;
final spring = motion.springSm; // motion-spring-sm (mass: 1.0, stiffness: 400, damping: 35)
final disableAnimations = MediaQuery.of(context).disableAnimations;

double scale = 1.0;
double opacity = 1.0;

void onPressedChanged(bool pressed) {
  if (disableAnimations) {
    scale = pressed ? 0.985 : 1.0;
    opacity = pressed ? 0.92 : 1.0;
    return;
  }

  controller.animateWith(
    SpringSimulation(
      spring,
      pressed ? 1.0 : 0.0,
      pressed ? 0.0 : 1.0,
      0,
    ),
  );
}

// disabled: estado final instantaneo
// skeleton/loading: usar el mismo spring para crossfade interno
