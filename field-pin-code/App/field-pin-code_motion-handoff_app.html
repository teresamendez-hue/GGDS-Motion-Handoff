Implementación Dart (Flutter)

final motion = context.motionTokens;
final spring = motion.springMd; // motion-spring-md (mass: 1.0, stiffness: 300, damping: 28)
final disableAnimations = MediaQuery.of(context).disableAnimations;

void onPinCellFocus(bool focused) {
  if (disableAnimations) return;
  controller.animateWith(SpringSimulation(spring, focused ? 1.0 : 0.0, focused ? 0.0 : 1.0, 0));
}
