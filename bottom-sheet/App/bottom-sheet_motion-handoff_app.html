Implementacion Dart (Flutter)

const spring = SpringDescription(mass: 1.0, stiffness: 300, damping: 28);
final disableAnimations = MediaQuery.of(context).disableAnimations;

void animateBottomSheetVisibility({
  required AnimationController controller,
  required bool visible,
}) {
  if (disableAnimations) {
    controller.value = visible ? 1.0 : 0.0;
    return;
  }
  controller.animateWith(
    SpringSimulation(spring, controller.value, visible ? 1.0 : 0.0, 0.0),
  );
}
